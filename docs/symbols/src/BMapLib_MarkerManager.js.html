 93     MarkerManager.prototype.addMarker = function (marker, minZoom, maxZoom) {
 94         minZoom = minZoom && minZoom > 0 ? minZoom : 1;
 95         maxZoom = maxZoom && maxZoom <= 19 ? maxZoom : this._opts.maxZoom;
 96         marker.minZoom = minZoom;
 97         marker.maxZoom = maxZoom;
 98         marker.bAdded = false;

220         _showMarkers: function () {
221             var num = this._numMarkers.length,
222                 curZoom = this._map.getZoom(),
223                 t = this._numMarkers,
224                 curBounds = this._getRealBounds();
225             for (var i = num; i--;) {
226                 //在可视区域内 && 当前zoom符合marker的显示条件

251             var curBounds = this._map.getBounds(),
252                 southWest = this._map.pointToPixel(curBounds.getSouthWest()),
253                 northEast = this._map.pointToPixel(curBounds.getNorthEast()),
254                 extendSW = {

258                 extendNE = {

260                     y: northEast.y - this._opts.borderPadding
261                 },
262                 extendSwPoint = this._map.pixelToPoint(extendSW),
263                 extendNePoint = this._map.pixelToPoint(extendNE);
264             return new BMap.Bounds(extendSwPoint, extendNePoint);
265         },
